CREATE TABLE ANIMAL (
    ID INTEGER PRIMARY KEY AUTOINCREMENT,
    NAME TEXT NOT NULL,
    CIENTIFIC_NAME TEXT NOT NULL,
    FAMILY TEXT CHECK(FAMILY IN ('carnivore', 'omnivore', 'herbivore')) NOT NULL,
    DANGER_OF_EXTINCTION BOOLEAN NOT NULL CHECK (DANGER_OF_EXTINCTION IN (0, 1)),
    SPECIES TEXT,
    GENDER TEXT,
    BIRTHDATE DATE,
    ORIGIN_COUNTRY TEXT,
    CONTINENT TEXT
);

CREATE TABLE ZOO (
    ID INTEGER PRIMARY KEY AUTOINCREMENT,
    ZOO_NAME TEXT NOT NULL,
    CITY TEXT,
    COUNTRY TEXT,
    ZOO_SIZE REAL,
    ANNUAL_BUDGET REAL,
    ANIMAL_ID INTEGER,
    FOREIGN KEY (ANIMAL_ID) REFERENCES ANIMAL(ID) ON DELETE SET NULL ON UPDATE CASCADE
);
