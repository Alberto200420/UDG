CREATE TABLE EXIT (
    ID INTEGER PRIMARY KEY AUTOINCREMENT,
    EXIT_DATE DATE NOT NULL,
    ENTRY_DATE DATE,
    DESTINATION TEXT
);

CREATE TABLE PARTNER (
    ID INTEGER PRIMARY KEY AUTOINCREMENT,
    NAME TEXT NOT NULL,
    LASTNAME TEXT NOT NULL
);

CREATE TABLE SHIP (
    ID INTEGER PRIMARY KEY AUTOINCREMENT,
    BRAND TEXT NOT NULL,
    MODEL TEXT,
    CAPACITY INTEGER,
    PRICE REAL,
    MOORING_NUMBER TEXT,
    SHARE REAL,
    PARTNER_ID INTEGER,
    EXIT_ID INTEGER,
    FOREIGN KEY (PARTNER_ID) REFERENCES PARTNER(ID) ON DELETE SET NULL ON UPDATE CASCADE,
    FOREIGN KEY (EXIT_ID) REFERENCES EXIT(ID) ON DELETE SET NULL ON UPDATE CASCADE
);
